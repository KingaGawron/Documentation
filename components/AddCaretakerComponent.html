import { ChangeDetectorRef, Component, OnInit, ViewChild } from '@angular/core';
import { FormBuilder, FormControl, FormGroup, NgForm, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { TabsetComponent } from 'ngx-bootstrap/tabs';
import { Caretaker } from 'src/app/model/icaretaker';
import { CaretakerService } from 'src/app/services/caretaker.service';
import { ICaretakerBase } from 'src/app/model/icaretakerbase';
import { MatTabGroup } from '@angular/material/tabs';
import { Ikeyvaluepair } from 'src/app/model/ikeyvaluepair';

@Component({
  selector: 'app-add-caretaker',
  templateUrl: './add-caretaker.component.html',
  styleUrls: ['./add-caretaker.component.css']
})
export class AddCaretakerComponent implements OnInit {
  @ViewChild('formTabs', { static: false }) formTabs?:MatTabGroup ;
  addCaretakerForm: FormGroup;
  nextClicked: boolean;
  buttonColor: string = '#E1DFE0'; 
  buttonNameColor: string = '#8D93A0';
  caretaker = new Caretaker();


  dogCatOther: Ikeyvaluepair[];
  offer: Ikeyvaluepair[];
  amenities: Ikeyvaluepair[];


  caretakerView: ICaretakerBase = {
    id: null,
    name: '',
    price: null,
    age: null,
    Type: null,
    city:null,
    description:null,
    address:null,
    dogCatOther:null,
    offer:null,
    moreThanOne:null,
    immediateAvailability:null,
    postedBy:null
  };
  immediateAvailabilitySelected(value: boolean) {
    this.caretaker.immediateAvailability = value;
  }
  constructor( private fb: FormBuilder,
    private router: Router,
    private caretakerService: CaretakerService,) { }

  ngOnInit() {
    this.CreateAddCaretakerForm();
    this.caretakerService.getAmenities().subscribe(data => {
      this.amenities=data;
    })
    this.caretakerService.getOffer().subscribe(data => {
      this.offer=data;
    })
    this.caretakerService.getDogCatOther().subscribe(data => {
      this.dogCatOther=data;
    })
  }
  CreateAddCaretakerForm() {
    this.addCaretakerForm = this.fb.group({
      BasicInfo: this.fb.group({
        Name: [null, Validators.required],
        City: [null, Validators.required],
        Address: [null, Validators.required],
        Age: [null, Validators.required],
        Price: [null, Validators.required],
        Description: [null, Validators.required],
        ImmediateAvailability:[null,Validators.required]

      }),

      OtherInfo: this.fb.group({
        MoreThanOne: [null],
        Offer: [null],
        Amenities: [null],
        Phone: [null],
        Email: [null],
        Cat: [false],
        Dog: [false],
        Other: [false],
        DogWalk:[false],
        QuickFeeding:[false],
        Daycare:[false],
        AllowedOnTheCouch: [false],
        CityLocation: [false],
        Apartment: [false],
        NoSmokingIndividuals: [false],
        AllowedOnTheBed: [false],
        NoSmallChildren: [false],
        NoOtherCats: [false],
        NoToxicPlants: [false],
        NoOtherDogs: [false],
        CloseToParks: [false],
        NoCages: [false]
        
      })
      });
      console.log('BasicInfo errors:', this.BasicInfo.errors);
  console.log('OtherInfo errors:', this.OtherInfo.errors);
  }
  get BasicInfo() {
    return this.addCaretakerForm.controls['BasicInfo'] as FormGroup;

    return this.addCaretakerForm.controls['OtherInfo'] as FormGroup;

  onBack() {
    this.router.navigate(['/']);
  }

  onSubmit() {
    this.nextClicked = true;
    if (this.allTabsValid()) {
      console.log("All tabs are valid.");
      this.mapCaretaker();
      this.caretakerService.addCaretaker(this.caretaker).subscribe(
        () => {
          console.log("Caretaker added successfully.");
          this.router.navigate(['/']);
        });
        //  else {
        //    console.log("Some tabs are invalid.");
        //  }
      
    //   console.log("Added");
    //   console.log(this.addCaretakerForm);
    //   this.router.navigate(['/']);
    // } else {
    //   console.log("Some tabs are invalid.");
    }
  }
  

  mapCaretaker(): void {
    this.caretaker.id = this.caretakerService.newCareID();
    this.caretaker.name = this.Name.value;
    this.caretaker.city = this.City.value;
    this.caretaker.address = this.Address.value;
    this.caretaker.age = this.Age.value;
    this.caretaker.price = this.Price.value;
    this.caretaker.description = this.Description.value;
    this.caretaker.immediateAvailability = this.ImmediateAvailability.value;
    this.caretaker.moreThanOne = this.MoreThanOne.value;
    if (this.OtherInfo.get('MoreThanOne').value === '1') {
      this.caretaker.moreThanOne = true;
    } else {
      this.caretaker.moreThanOne = false;
    }
    if (this.BasicInfo.get('ImmediateAvailability').value === '1') {
      this.caretaker.immediateAvailability = true;
    } else {
      this.caretaker.immediateAvailability = false;
    }
    this.caretaker.dogCatOther = {
    cat: this.OtherInfo.get('Cat').value,
    dog: this.OtherInfo.get('Dog').value ,
    other: this.OtherInfo.get('Other').value
  };
    this.caretaker.offer = {
      dogWalk: this.OtherInfo.get('DogWalk').value,
      quickFeeding: this.OtherInfo.get('QuickFeeding').value,
      daycare: this.OtherInfo.get('Daycare').value
    }
    this.caretaker.amenities = {
      allowedOnTheCouch: this.OtherInfo.get('AllowedOnTheCouch').value,
      cityLocation: this.OtherInfo.get('CityLocation').value,
      apartment: this.OtherInfo.get('Apartment').value,
      noSmokingIndividuals: this.OtherInfo.get('NoSmokingIndividuals').value,
      allowedOnTheBed: this.OtherInfo.get('AllowedOnTheBed').value,
      noSmallChildren: this.OtherInfo.get('NoSmallChildren').value,
      noOtherCats: this.OtherInfo.get('NoOtherCats').value,
      noToxicPlants: this.OtherInfo.get('NoToxicPlants').value,
      noOtherDogs: this.OtherInfo.get('NoOtherDogs').value,
      closeToParks: this.OtherInfo.get('CloseToParks').value,
      noCages: this.OtherInfo.get('NoCages').value
    };
    
    this.caretaker.phone = this.Phone.value;
    this.caretaker.email = this.Email.value;
    this.caretaker.PostedOn = new Date().toString();
  }

  allTabsValid(): boolean {
    if (this.BasicInfo.invalid) {
      this.formTabs.selectedIndex = 0;
      return false;
    }

    if (this.OtherInfo.invalid) {
      this.formTabs.selectedIndex = 1;
      return false;
    }
    return true;
  }

  selectTab(NextTabId: number, IsCurrentTabValid: boolean) {
    this.nextClicked = true;
    this.formTabs.selectedIndex = NextTabId;

    if (IsCurrentTabValid) {

      this.buttonColor = '#E79DAC';
      this.buttonNameColor = '#FFFFFF';

      
    } else {
      console.log('Current tab is invalid. Please fill in all required fields.');
      this.buttonColor = '#E1DFE0';
      this.buttonNameColor = '#8D93A0';
    }
  }
  
}
