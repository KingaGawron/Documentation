import { Component, OnInit } from '@angular/core';
import { ICaretakerBase } from 'src/app/model/icaretakerbase';
import { CaretakerService } from 'src/app/services/caretaker.service';

@Component({
  selector: 'app-caretakers-list',
  templateUrl: './caretakers-list.component.html',
  styleUrls: ['./caretakers-list.component.css'],
})
export class CaretakersListComponent implements OnInit {
  allCaretakers!: Array<ICaretakerBase>; 
  caretakers!: Array<ICaretakerBase>;
  City = '';
  SearchCity = '';
  SortbyParam = '';
  SortDirection = 'asc';
  // maxPrice = 400;

  DogFilter = false;
  CatFilter = false;
  OtherFilter = false;
  DogWalkFilter = false;
  QuickFeedingFilter = false;
  DaycareFilter = false;
  selectedMultiplePets: boolean | null = null;
  immediateAvailabilityFilter = false;


  constructor(private caretakerService: CaretakerService) {}

  ngOnInit(): void {
    this.caretakerService.getAllCaretakers().subscribe((data) => {
      this.allCaretakers = data;
      this.caretakers = [...this.allCaretakers]; 
    });
  }

  onCityFilter() {
    this.SearchCity = this.City;
    this.onFilter();
  }

  onFilter() {
    this.caretakers = this.allCaretakers.filter((caretaker) => {
      return (
        (this.SearchCity === '' || caretaker.city.toLocaleLowerCase() === this.SearchCity.toLocaleLowerCase()) &&
        (!this.DogFilter || caretaker.dogCatOther.dog) &&
        (!this.CatFilter || caretaker.dogCatOther.cat) &&
        (!this.OtherFilter || caretaker.dogCatOther.other) &&
        (!this.DogWalkFilter || caretaker.offer.dogWalk) &&
        (!this.QuickFeedingFilter || caretaker.offer.quickFeeding) &&
        (!this.DaycareFilter || caretaker.offer.daycare) &&
        (!this.selectedMultiplePets || caretaker.moreThanOne === this.selectedMultiplePets) &&
        (!this.immediateAvailabilityFilter || caretaker.immediateAvailability) 
      );
    });
  }

  onCityFilterClear() {
    this.SearchCity = '';
    this.City = '';
    this.DogFilter = false;
    this.CatFilter = false;
    this.OtherFilter = false;
    this.DogWalkFilter = false;
    this.QuickFeedingFilter = false;
    this.DaycareFilter = false;
    this.caretakers = [...this.allCaretakers];
  }

  onSortDirection() {
    if (this.SortbyParam !== '') {
      this.caretakers.sort((a, b) => {
        const aValue = a[this.SortbyParam];
        const bValue = b[this.SortbyParam];
  
        if (this.SortDirection === 'asc') {
          return aValue < bValue ? -1 : aValue > bValue ? 1 : 0;
        } else {
          return bValue < aValue ? -1 : bValue > aValue ? 1 : 0;
        }
      });
    }
  }
  sliderValue = 0;
}
