import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import { Caretaker } from 'src/app/model/icaretaker';
import { Photo } from 'src/app/model/photo';
import { CaretakerService } from 'src/app/services/caretaker.service';

@Component({
  selector: 'app-caretaker-detail',
  templateUrl: './caretaker-detail.component.html',
  styleUrls: ['./caretaker-detail.component.css']
})



export class CaretakerDetailComponent implements OnInit {

editPhoto() {
throw new Error('Method not implemented.');
}




  public caretakerId!: number;
  caretaker: Caretaker = new Caretaker(); 
  // public mainPhotoUrl: string = null;
  // galleryOptions: NgxGalleryOptions[];
  // galleryImages: NgxGalleryImage[];
  showOptions: boolean = false;


  constructor(private route: ActivatedRoute, private caretakerService: CaretakerService) { }

  ngOnInit() {
    this.route.params.subscribe(
      (params) => {
        this.caretakerId = +params['id'];
        this.caretakerService.getCaretaker(this.caretakerId).subscribe(
          (data: Caretaker) => {
            this.caretaker = data;
          }
        );
      }
    );
  }
  

  ratingcount=0;
  totalrating=0

  Finalrating:any;

  ratingcontrol=new FormControl(0);
  GetRating(){
    this.ratingcount++;
    this.totalrating +=this.ratingcontrol?.value || 0;
    this.Finalrating= (this.totalrating/this.ratingcount).toFixed(2)

  }
}
