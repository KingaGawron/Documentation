import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { map } from 'rxjs/operators';
import { Observable } from 'rxjs';
import { Caretaker } from '../model/icaretaker';
import { ICaretakerBase } from '../model/icaretakerbase';
import { environment } from 'src/assets/environments/environments';
import { Ikeyvaluepair } from '../model/ikeyvaluepair';

@Injectable({
  providedIn: 'root',
})
export class CaretakerService {
  baseUrl = environment.baseUrl;

  constructor(private http: HttpClient) {}
  getCaretaker(id: number) {
    return this.http.get<Caretaker>(
      this.baseUrl + '/caretaker/detail/' + id.toString()
    );

   
  }
  getOffer(): Observable<Ikeyvaluepair[]> {
    return this.http.get<Ikeyvaluepair[]>(this.baseUrl + '/offer/list');
}
getDogCatOther(): Observable<Ikeyvaluepair[]> {
  return this.http.get<Ikeyvaluepair[]>(this.baseUrl + '/dogcatother/list');
}

getAmenities(): Observable<Ikeyvaluepair[]> {
    return this.http.get<Ikeyvaluepair[]>(this.baseUrl + '/amenities/list');
}
  getAllCaretakers(): Observable<ICaretakerBase[]> {
    return this.http.get<Caretaker[]>(this.baseUrl + '/caretaker/list');

  }
  addCaretaker(caretaker: Caretaker) {
    const httpOptions = {
      headers: new HttpHeaders({
          Authorization: 'Bearer '+ localStorage.getItem('token')
      })
  };
  return this.http.post(this.baseUrl + '/caretaker/add', caretaker,httpOptions);

   
  }
  
  
  newCareID() {
    if (localStorage.getItem('CID')) {
      localStorage.setItem('CID', String(+localStorage.getItem('CID') + 1));
      return +localStorage.getItem('CID');
    } else {
      localStorage.setItem('CID', '101');
      return 101;
    }
  }
getCaretakerIdByPostedBy(postedById: number): Observable<number> {
  return this.http.get<number>(`${this.baseUrl}/caretaker/detail/${postedById}`);
}

}
